\section{v8 Genus Configuration (5M Balanced)}

\begin{lstlisting}[caption={YAML configuration for v8 genus experiment.}, label={lst:v8-config}]
model:
  backbone: InstaDeepAI/nucleotide-transformer-v2-500m-multi-species
  head_type: attention_pool
  num_query_tokens: 4
  hidden_dim: 512
  dropout: 0.15

lora:
  r: 16
  alpha: 32
  dropout: 0.05
  target_modules: [query, key, value]

data:
  fasta_path: /path/to/balanced_5M/reads.fa
  labels_path: /path/to/balanced_5M/labels.tsv
  classification_level: genus
  max_length: 32
  val_ratio: 0.1

training:
  batch_size: 32
  gradient_accumulation_steps: 2
  num_epochs: 20
  phase1_epochs: 5
  backbone_lr: 3.0e-5
  head_lr: 5.0e-4
  weight_decay: 0.02
  warmup_ratio: 0.1
  early_stopping_patience: 7
  rc_augmentation: true
\end{lstlisting}

\section{SLURM Job Script Example}

\begin{lstlisting}[language=bash, caption={SLURM script for training on TWCC cluster.}, label={lst:slurm}]
#!/bin/bash
#SBATCH --job-name=gfm-genus-v9
#SBATCH --partition=gpu
#SBATCH --gres=gpu:1
#SBATCH --cpus-per-task=4
#SBATCH --mem=128G
#SBATCH --time=48:00:00
#SBATCH --output=slurm-%j.out

module load cuda/12.1
source activate gfm

python scripts/train.py \
  --config configs/nt_token_genus_v9_50M.yaml \
  --init_from results/nt_token_genus_lora_v8_5M/best.pt

python scripts/evaluate.py \
  --config configs/nt_token_genus_v9_50M.yaml \
  --checkpoint results/nt_token_genus_lora_v9_50M/best.pt

python scripts/evaluate.py \
  --config configs/nt_token_genus_v9_50M.yaml \
  --checkpoint results/nt_token_genus_lora_v9_50M/best.pt \
  --rc_tta
\end{lstlisting}
